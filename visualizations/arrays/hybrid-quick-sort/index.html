def _partition(nums, l, h):
    pivot = nums[random.randint(l, h)]
    low, mid, high = l, l, h
    while mid <= high:
        if nums[mid] < pivot:
            swap(nums[low], nums[mid])
            low += 1; mid += 1
        elif nums[mid] == pivot:
            mid += 1
        else: # nums[mid] > pivot
            swap(nums[mid], nums[high])
            high -= 1
    return low, high
 
def _quickSort(nums, l, h):
    if not l < h: return
    low, high = _partition(nums, l, h)
    _quickSort(nums, l, low - 1)
    _quickSort(nums, high + 1, h)
